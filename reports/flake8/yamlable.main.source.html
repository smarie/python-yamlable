55 class YamlAble(AbstractYamlAble):
56 """
57 A helper class to register a class as able to dump instances to yaml and to load them back from yaml.
58  
59 This class does not rely on the `YAMLObject` class provided in pyyaml, so it provides a bit more flexibility (no
60 metaclass magic).
61  
62 The behaviour is very similar though:
63 - inherit from `YamlAble` or virtually inherit from it using YamlAble.register(cls)
64 - fill the `__yaml_tag_suffix__` either directly or using the `@yaml_info()` decorator
65 - optionally override `__from_yaml_dict__` (class method called during decoding) and/or `__to_yaml_dict__`
66 (instance method called during encoding) if you wish to have control on the process, for example to only dump part
67 of the attributes or perform some custom instance creation. Note that default implementation relies on `vars(self)`
68 for dumping and on `cls(**dct)` for loading.
69 """
70 __yaml_tag_suffix__ = None
71 """ placeholder for a class-wide yaml tag. It will be prefixed with '!yamlable/', stored in `YAMLABLE_PREFIX` """
72  
73 @classmethod
74 def is_yaml_tag_supported(cls,
75 yaml_tag # type: str
76 ):
77 # type: (...) -> bool
78 """
79 Implementing classes should return True if they are able to decode yaml objects with this yaml tag.
80 Default implementation relies on class attribute `__yaml_tag_suffix__` if provided, either manually or through
81 the `@yaml_info` decorator.
82  
83 :param yaml_tag:
84 :return:
85 """
86 if hasattr(cls, '__yaml_tag_suffix__') and cls.__yaml_tag_suffix__ is not None:
87 if '__yaml_tag_suffix__' in cls.__dict__:
88 # this is an explicitly configured class (__yaml_tag_suffix__ is set on it), ok
89 return cls.__yaml_tag_suffix__ == yaml_tag
90 else:
91 # this class inherits from the __yaml_tag_suffix__ and does not redefine it, not ok
92 raise TypeError("`__yaml_tag_suffix__` field is not redefined by class {}, cannot inherit from YamlAble"
93 "properly.".format(cls))
94  
95 else:
96 raise NotImplementedError("class {} does not seem to have a non-None '__yaml_tag_suffix__' field. You can "
97 "either create one manually or by decorating your class with @yaml_info. "
98 "Alternately you should override the 'is_yaml_tag_supported' method "
99 "from YamlAble.".format(cls))

102 def yaml_info(yaml_tag=None, # type: str
103 yaml_tag_ns=None # type: str
104 ):
105 # type: (...) -> Callable[[Type[YA]], Type[YA]]
106 """
107 A simple class decorator to tag a class with a global yaml tag - that way you do not have to call `YamlAble` super
108 constructor.
109  
110 You can either provide a full yaml tag suffix:
111  
112 ```python
113 @yaml_info("com.example.MyFoo")
114 class Foo(YamlAble):
115 pass
116  
117 print(Foo.__yaml_tag_suffix__) # yields "com.example.MyFoo"
118 ```
119  
120 or simply provide a namespace, that will be appended with '.<class name>' :
121  
122 ```python
123 @yaml_info(yaml_tag_ns="com.example")
124 class Foo(YamlAble):
125 pass
126  
127 print(MyFoo.__yaml_tag_suffix__) # yields "com.example.Foo"
128 ```
129  
130 In both cases, the suffix is appended at the end of the common yamlable prefix:
131  
132 ```python
133 print(Foo().dumps_yaml()) # yields "!yamlable/com.example.Foo {}"
134 ```
135  
136 :param yaml_tag: the complete yaml suffix.
137 :param yaml_tag_ns: the yaml namespace. It will be appended with '.<cls.__name__>'
138 :return:
139 """
140 def f(cls # type: Type[YA]
141 ):
142 # type: (...) -> Type[YA]
143 return yaml_info_decorate(cls, yaml_tag=yaml_tag, yaml_tag_ns=yaml_tag_ns)
144 return f

147 def yaml_info_decorate(cls, # type: Type[YA]
148 yaml_tag=None, # type: str
149 yaml_tag_ns=None # type: str
150 ):
151 # type: (...) -> Type[YA]
152 """
153 A simple class decorator to tag a class with a global yaml tag - that way you do not have to call `YamlAble` super
154 constructor.
155  
156 You can either provide a full yaml tag suffix:
157  
158 ```python
159 @yaml_info("com.example.MyFoo")
160 class Foo(YamlAble):
161 pass
162  
163 print(Foo.__yaml_tag_suffix__) # yields "com.example.MyFoo"
164 ```
165  
166 or simply provide a namespace, that will be appended with '.<class name>' :
167  
168 ```python
169 @yaml_info(yaml_tag_ns="com.example")
170 class Foo(YamlAble):
171 pass
172  
173 print(MyFoo.__yaml_tag_suffix__) # yields "com.example.Foo"
174 ```
175  
176 In both cases, the suffix is appended at the end of the common yamlable prefix:
177  
178 ```python
179 print(Foo().dumps_yaml()) # yields "!yamlable/com.example.Foo {}"
180 ```
181  
182 :param cls:
183 :param yaml_tag: the complete yaml suffix.
184 :param yaml_tag_ns: the yaml namespace. It will be appended with '.<cls.__name__>'
185 :return:
186 """
187 if yaml_tag_ns is not None:
188 if yaml_tag is not None:
189 raise ValueError("Only one of 'yaml_tag' and 'yaml_tag_ns' should be provided")
190  
191 # create yaml_tag by appending the class name to the namespace
192 yaml_tag = yaml_tag_ns + '.' + cls.__name__
193  
194 elif yaml_tag is None:
195 raise ValueError("One non-None `yaml_tag` or `yaml_tag_ns` must be provided.")
196  
197 if issubclass(cls, YamlObject2):
198 if not yaml_tag.startswith('!'):
199 raise ValueError("When extending YamlObject2, the `yaml_tag` field should contain the full yaml tag, "
200 "and should therefore start with !")
201 cls.yaml_tag = yaml_tag
202  
203 elif issubclass(cls, YamlAble):
204 if yaml_tag.startswith('!'):
205 raise ValueError("When extending YamlAble, the `yaml_tag` field should only contain the yaml tag suffix, "
206 "and should therefore NOT start with !")
207 cls.__yaml_tag_suffix__ = yaml_tag # type: ignore
208 else:
209 raise TypeError("classes tagged with @yaml_info should be subclasses of YamlAble or YamlObject2")
210  
211 return cls

215 def decode_yamlable(loader,
216 yaml_tag, # type: str
217 node, # type: MappingNode
218 **kwargs):
219 # type: (...) -> YamlAble
220 """
221 The method used to decode YamlAble object instances
222  
223 :param loader:
224 :param yaml_tag:
225 :param node:
226 :param kwargs:
227 :return:
228 """
229 candidates = _get_all_subclasses(YamlAble)
230 errors = dict()
231 for clazz in candidates:
232 try:
233 if clazz.is_yaml_tag_supported(yaml_tag):
234 constructor_args = read_yaml_node_as_dict(loader, node)
235 return clazz.__from_yaml_dict__(constructor_args, yaml_tag=yaml_tag)
236 except Exception as e:
237 errors[clazz.__name__] = e
238  
239 raise TypeError("No YamlAble subclass found able to decode object !yamlable/" + yaml_tag + ". Tried classes: "
240 + str(candidates) + ". Caught errors: " + str(errors) + ". "
241 "Please check the value of <cls>.__yaml_tag_suffix__ on these classes. Note that this value may be "
242 "set using @yaml_info() so help(yaml_info) might help too.")

257 :param kwargs:

261 new_data = obj.__to_yaml_dict__()

264 # TODO check that it works
265 return dumper.represent_mapping(None, new_data, flow_style=None)

362 class YamlCodec(six.with_metaclass(ABCMeta, object)):
363 """
364 Represents a codec class, able to encode several object types into/from yaml, with potentially different yaml tag
365 ids. It assumes that the objects are written as yaml dictionaries, and that they all have the same yaml tag prefix
366  
367 for example !mycodec/<yaml_tag_suffix>, where 'mycodec' is the yaml prefix associated with this codec.
368  
369 This allows the code to be pre-wired so that it is very easy to implement.
370 - Decoding:
371 - fill get_yaml_prefix
372 - fill is_yaml_tag_supported to declare if a given yaml tag is supported or not
373 - fill from_yaml_dict to create new instances of objects from a dictionary, according to the yaml tag
374 - Encoding:
375 - fill get_known_types
376 - fill the to_yaml_dict
377 """
378

450 @classmethod
451 def encode(cls, dumper, obj,
452 without_custom_tag=False, # type: bool
453 **kwargs):
454 # type: (...) -> MappingNode
455 """
456 The method used to encode YamlAble object instances
457  
458 :param dumper:
459 :param obj:
460 :param without_custom_tag: if this is set to True, the yaml tag !yamlable/<yaml_tag_suffix> will not be written
461 to the document. Warning: if you do so, decoding the object will not be easy!
462 :param kwargs: keyword arguments coming from pyyaml, not sure what you will find here.
463 :return:
464 """
465 # Convert objects to a dictionary of their representation
466 yaml_tag_suffix, obj_as_dict = cls.to_yaml_dict(obj)
467 if not isinstance(obj_as_dict, collections.Mapping) or not isinstance(yaml_tag_suffix, str):
468 raise TypeError("`to_yaml_dict` did not return correct results. It should return a tuple of "
469 "`yaml_tag_suffix, obj_as_dict`")
470  
471 if without_custom_tag:
472 # TODO check that it works
473 return dumper.represent_mapping(None, obj_as_dict, flow_style=None)
474 else:
475 # Add the tag information
476 prefix = cls.get_yaml_prefix()
477 if len(prefix) == 0 or prefix[-1] != '/':
478 prefix = prefix + '/'
479 yaml_tag = prefix + yaml_tag_suffix
480 return dumper.represent_mapping(yaml_tag, obj_as_dict, flow_style=None)

495 @classmethod
496 def register_with_pyyaml(cls, loaders=ALL_PYYAML_LOADERS, dumpers=ALL_PYYAML_DUMPERS):
497 # type: (...) -> None
498 """
499 Registers this codec with PyYaml, on the provided loaders and dumpers (default: all PyYaml loaders and dumpers).
500 - The encoding part is registered for the object types listed in cls.get_known_types(), in order
501 - The decoding part is registered for the yaml prefix in cls.get_yaml_prefix()
502  
503 :param loaders: the PyYaml loaders to register this codec with. By default all pyyaml loaders are considered
504 (Loader, SafeLoader...)
505 :param dumpers: the PyYaml dumpers to register this codec with. By default all pyyaml loaders are considered
506 (Dumper, SafeDumper...)
507 :return:
508 """
509 for loader in loaders:
510 loader.add_multi_constructor(cls.get_yaml_prefix(), cls.decode)
511  
512 for dumper in dumpers:
513 for t in cls.get_known_types():
514 dumper.add_multi_representer(t, cls.encode)
